def find_median(data_arr):
    sorted_arr = sorted(data_arr)
    while len(sorted_arr) > 2:
        sorted_arr = sorted_arr[1:-1]

    if len(sorted_arr) == 2:
        return (sorted_arr[0] + sorted_arr[1]) / 2.0
